🔗 Connect to Google Sheets

This links the archive checkboxes to a shared Google Sheet so everyone on the team sees the same selections in real time. Setup takes about 3 minutes.

  1. Open the Google Sheet that was created alongside this tool: SUP Confluence Audit — Page Inventory
  2. In the sheet, go to Extensions → Apps Script
  3. Delete any existing code and paste in the script below, then click Save
  4. Click Deploy → New deployment, set type to Web app, set access to "Anyone" or "Anyone within [your org]", click Deploy
  5. Copy the Web app URL and paste it below
function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = sheet.getDataRange().getValues();
  var archived = [];
  for (var i = 1; i < data.length; i++) {
    if (data[i][8] === true || data[i][8] === "TRUE") {
      archived.push(String(data[i][0]));
    }
  }
  return ContentService
    .createTextOutput(JSON.stringify({archived: archived}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var params = JSON.parse(e.postData.contents);
  var data = sheet.getDataRange().getValues();
  var idCol = 0, archiveCol = 8;
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][idCol]) === String(params.id)) {
      sheet.getRange(i + 1, archiveCol + 1).setValue(params.archived);
      break;
    }
  }
  return ContentService
    .createTextOutput(JSON.stringify({ok: true}))
    .setMimeType(ContentService.MimeType.JSON);
}
